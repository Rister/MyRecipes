\begin{recipe}{Butter Tart Squares}{}{}
    \newstep
    Line a 9\X 13 pan with parchment paper and spray with non stick spray.

    Preheat oven to 350\0 .

    \ingredient[2]{cups}{all purpose flour}
    \ingredient[4]{tbsp}{powdered icing sugar}
    In a large bowl, combine flour and powdered sugar.

    \ingredient[1]{cup}{unsalted butter (cold, cubed)}
    \ingredient[1--2]{tbsp}{cold water}
    Cut butter into icing sugar and flour with a pastry cutter or your hands until pea-sized lumps of butter remain.
    When you grab a handful and squeeze, you want it to hold together.
    If it doesn't hold together, add 1--2 tablespoons of water.

    \newstep
    Dump flour mixture into pan and press down evenly to form a crust.

    Bake for 10 minutes.

    \ingredient[\fr{1}{2}]{cup}{butter (unsalted)}
    Meanwhile, prepare your filling.

    In a medium bowl melt butter in the microwave.

    \ingredient[3]{cups}{brown sugar}
    Whisk in brown sugar until smooth.

    \ingredient[4]{}{large eggs}
    \ingredient[1--2]{tbsp}{white vinegar}
    \ingredient[1]{tbsp}{vanilla}
    \ingredient[\fr{1}{4}]{cup}{raisins (optional)}
    Whisk in eggs, one at a time, then add the vinegar and vanilla.

    \newstep
    Pour over crust and bake another 25-30 minutes until filling is set or almost set --- a slight jiggle is fine but you don't want a lot of movement.

    Cool to room temperature and refrigerate several hours before slicing.
\end{recipe}

\begin{recipe}{Snickerdoodle Bar Cookies}{}{}
    \newstep
    Preheat oven to 350\0 F and spray an 8-inch square metal baking pan lightly with nonstick spray.
    Line the pan with parchment paper.

    \ingredient[1\fr{1}{4}]{cups}{all-purpose flour}
    \ingredient[2]{tsp}{cream of tartar}
    \ingredient[\fr{1}{2}]{tsp}{baking powder}
    \ingredient[\fr{1}{8}]{tsp}{baking soda}
    \ingredient[2]{tsp}{ground cinnamon}
    \ingredient[\fr{1}{2}]{tsp}{salt}
    Whisk together flour, cream of tartar, baking powder, baking soda, cinnamon, and salt in a medium bowl.

    \ingredient[1]{cup}{unsalted butter (room temperature)}
    \ingredient[1]{cup}{granulated sugar}
    \ingredient[\fr{1}{4}]{cup}{light brown sugar}
    Beat butter and sugars in a large mixer bowl at medium speed until creamy (about 2 minutes).

    \ingredient[1]{}{large egg}
    \ingredient[1]{}{large egg yolk}
    \ingredient[1\fr{1}{2}]{tsp}{vanilla extract}
    Beat in egg, egg yolkk, and vanilla until combined.

    \newstep
    Reduce speed to low and gradually add dry ingredients; beat just until incorporated (the dough will be soft).

    \newstep
    Scrape dough into prepared pan and use a piece of parchment paper or wax paper to press dough evenly into pan.
    Gently tap pan once to remove air bubbles.

    \ingredient[2]{tbsp}{granulated sugar}
    \ingredient[\fr{1}{2}]{tsp}{ground cinnamon}
    Combine sugar and cinnamon in a small bowl, and sprinkle topping evenly over the snickerdoodle dough.

    \newstep
    Bake bars for 24 to 25 minutes, until edges are golden, top is set, and a toothpick inserted in the center comes out clean. Cool completely in baking pan on a wire rack.

    Lift parchment and place bars ona cutting board.
    Cut into squares with a sharp knife.
    Store leftover cookie bars in an airtight container at room temperature.
\end{recipe}

\begin{recipe}{Grandma's Corn Pudding}{}{}

    \newstep
    Preheat the oven to 400\0 F (200\0 C). Grease a 2-quart casserole dish.

    \ingredient[5]{}{large eggs}
    Whisk eggs lightly in a large bowl.

    \ingredient[\fr{1}{3}]{cup}{butter}
    Melt the butter and let cool slightly.

    \ingredient[\fr{1}{2}]{cup}{milk}
    \ingredient[\fr{1}{4}]{cup}{white sugar}
    \ingredient[\fr{1}{4}]{cup}{cornstarch}
    Add milk, melted butter, sugar, and cornstarch; whisk until well combined.

    \ingredient[1]{can (15.25 oz)}{whole kernel corn}
    \ingredient[2]{cans (14.75 oz)}{cream-style corn}
    Stir in drained corn and cream-style corn until fully blended.

    \newstep
    Pour mixture into the prepared casserole dish.

    \newstep
    Bake in the preheated oven until golden brown on top, about 1 hour.
\end{recipe}

\begin{recipe}{Ham Loaf}{}{}
    \newstep
    Preheat the oven to 350\0 F (175\0 C).

    \Ingredient{Cooking Spray}
    Lightly grease a 9x13-inch baking dish with cooking spray.

    \ingredient[2]{pounds}{ground ham}
    \ingredient[1\fr{1}{2}]{pounds}{ground pork}
    \ingredient[1]{cup}{dried bread crumbs}
    \ingredient[1]{cup}{evaporated milk}
    \Ingredient{2 large eggs}
    \ingredient[\fr{1}{8}]{teaspoon}{salt}
    \ingredient[\fr{1}{8}]{teaspoon}{ground black pepper}
    Combine ham, pork, bread crumbs, evaporated milk, eggs, salt and pepper in a large bowl; mix well.
    Form into a loaf and place into the prepared baking dish.

    \newstep
    Bake in the preheated oven for 1 hour 15 minutes.

    \ingredient[1]{cup}{brown sugar}
    \ingredient[\fr{1}{4}]{cup}{cider vinegar}
    \ingredient[1]{tablespoon}{mustard powder}
    While loaf is baking, make glaze.
    Stir together brown sugar, vinegar, and mustard powder in a small bowl until combined.

    \newstep
    Pour glaze over loaf and continue baking until an instant-read thermometer inserted into the center of loaf reads at least 160\0 F (72\0 C), about 15 minutes more.
\end{recipe}
