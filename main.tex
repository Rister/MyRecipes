\begin{recipe}{Butter Tart Squares}{}{}
  \newstep
  Line a 9x13-inch pan with parchment paper and spray with nonstick spray.

  Preheat oven to 350\0 F.

  \ing[2]{cups}{all-purpose flour}
  \ing[4]{tbsp}{powdered sugar}
  In a large bowl, combine the flour and powdered sugar.

  \ing[1]{cup}{unsalted butter (cold, cubed)}
  \ing[1--2]{tbsp}{cold water}
  Cut the butter into the combined powdered sugar and flour with a
  pastry cutter or your
  hands until pea-sized lumps of butter remain.
  The mixture should hold together when squeezed.
  If it doesn't hold together, add 1 to 2 tablespoons of cold water.

  \newstep
  Press the flour mixture evenly into the pan to form a crust.

  Bake for 10 minutes.

  \ing[\fr{1}{2}]{cup}{unsalted butter}
  Meanwhile, prepare the filling:

  In a medium bowl, melt the butter in the microwave.

  \ing[3]{cups}{brown sugar}
  Whisk in the brown sugar until smooth.

  \ing[4]{}{large eggs}
  \ing[1--2]{tbsp}{white vinegar}
  \ing[1]{tbsp}{vanilla}
  \ing[\fr{1}{4}]{cup}{raisins (optional)}
  Whisk in the eggs, one at a time, then add the vinegar and vanilla.

  \newstep
  Pour over the crust and bake for another 25 to 30 minutes, or until
  the filling is set
  or almost set. A slight jiggle is acceptable, but avoid excessive movement.

  Cool to room temperature, then refrigerate for several hours before slicing.
\end{recipe}

\begin{recipe}{Snickerdoodle Bar Cookies}{}{}
  \newstep
  Preheat oven to 350\0 F and lightly spray an 8x8-inch metal baking pan
  with nonstick spray.
  Line the pan with parchment paper.

  \ing[1\fr{1}{4}]{cups}{all-purpose flour}
  \ing[2]{tsp}{cream of tartar}
  \ing[\fr{1}{2}]{tsp}{baking powder}
  \ing[\fr{1}{8}]{tsp}{baking soda}
  \ing[2]{tsp}{ground cinnamon}
  \ing[\fr{1}{2}]{tsp}{salt}
  In a medium bowl, whisk together the flour, cream of tartar, baking
  powder, baking soda,
  cinnamon, and salt.

  \ing[1]{cup}{unsalted butter (room temperature)}
  \ing[1]{cup}{granulated sugar}
  \ing[\fr{1}{4}]{cup}{light brown sugar}
  In a large mixer bowl, beat the butter and sugars at medium speed until
  creamy (about 2 minutes).

  \ing[1]{}{large egg}
  \ing[1]{}{large egg yolk}
  \ing[1\fr{1}{2}]{tsp}{vanilla extract}
  Beat in the egg, egg yolk, and vanilla until combined.

  \newstep
  Reduce speed to low and gradually add the dry ingredients; beat just
  until incorporated (the dough will be soft).

  \newstep
  Scrape the dough into the prepared pan and use a piece of parchment
  or wax paper
  to press it evenly into the pan.
  Gently tap the pan once to remove air bubbles.

  \ing[2]{tbsp}{granulated sugar}
  \ing[\fr{1}{2}]{tsp}{ground cinnamon}
  In a small bowl, combine the sugar and cinnamon, then sprinkle the topping
  evenly over the snickerdoodle dough.

  \newstep
  Bake the bars for 24 to 25 minutes, until the edges are golden, the
  top is set,
  and a toothpick inserted in the center comes out clean. Cool
  completely in the baking pan on a wire rack.

  Lift the parchment and place the bars on a cutting board.
  Cut into squares with a sharp knife.
  Store leftover cookie bars in an airtight container at room temperature.
\end{recipe}

\begin{recipe}{Ham Loaf}{}{}
  \newstep
  Preheat the oven to 350\0 F (175\0 C).

  \Ing{cooking spray}
  Lightly grease a 9x13-inch baking dish with cooking spray.

  \ing[2]{lbs}{ground ham}
  \ing[1\fr{1}{2}]{lbs}{ground pork}
  \ing[1]{cup}{dried bread crumbs}
  \ing[1]{cup}{evaporated milk}
  \ing[2]{}{large eggs}
  \ing[\fr{1}{8}]{tsp}{salt}
  \ing[\fr{1}{8}]{tsp}{ground black pepper}
  In a large bowl, combine the ground ham, ground pork, dried bread
  crumbs, evaporated milk, eggs, salt, and
  pepper; mix well.
  Form the mixture into a loaf and place it into the prepared baking dish.

  \newstep
  Bake in the preheated oven for 1 hour 15 minutes.

  \ing[1]{cup}{brown sugar}
  \ing[\fr{1}{4}]{cup}{cider vinegar}
  \ing[1]{tbsp}{mustard powder}
  While the loaf is baking, prepare the glaze.
  In a small bowl, stir together the brown sugar, cider vinegar, and
  mustard powder until combined.

  \newstep
  Pour the glaze over the loaf and continue baking for about 15
  minutes more, or until an instant-read
  thermometer inserted into the center of the loaf reads at least 160\0 F
  (72\0 C).
\end{recipe}

\begin{recipe}{Corn Gazpacho}{}{}
  \ing[3]{cups}{fresh corn kernels}

  Measure out a few tablespoons of corn kernels and set them aside
  for garnish, if desired.

  \ing[1]{}{yellow bell pepper}
  \ing[\fr{1}{2}]{}{english cucumber, peeled and roughly chopped}
  \ing[1\fr{1}{2}]{cups}{yellow or orange cherry tomatoes}
  \ing[2]{tbsp}{apple cider vinegar}
  \ing[2]{tbsp}{olive oil}
  \ing[\fr{2}{3}]{cups}{cold water}
  \ing[1]{tsp}{kosher salt}
  \ing[\fr{1}{2}]{tsp}{freshly ground black pepper}

  Add the remaining corn, chopped bell pepper, chopped cucumber, cherry
  tomatoes, vinegar, oil, water, salt, and pepper to a blender.
  Blend on high speed until very smooth (2 to 3 minutes).

  \newstep

  Taste the soup and season with more salt, pepper, and vinegar as
  desired. Then, remove the blender jar from the base and transfer to
  the refrigerator to chill for at least 2 hours.

  \Ing{ice cubes}

  Once the gazpacho has chilled thoroughly, pour or ladle it into
  serving bowls or glasses. Add an ice cube to each serving, if desired.
  Top with the reserved corn kernels (if using), a drizzle of
  olive oil, and some freshly cracked black pepper, then serve.

\end{recipe}

\begin{recipe}{No-Bake PB\&J Bars}{}{}

  \ing[3\fr{1}{2}]{cups}{mini pretzels}

  Process the pretzels in a food processor until small crumbs form (15 to
  30 seconds).

  \ing[\fr{1}{4}]{cup}{white sugar}
  \ing[9]{tbsp}{melted unsalted butter}
  \ing[\fr{1}{4}]{tsp}{kosher salt}

  Add the sugar, butter, and salt; pulse until evenly combined (about 5 pulses).

  \newstep

  Transfer the pretzel mixture to an 8-inch square baking dish and press
  firmly into an even layer.
  Chill, uncovered, until firm (about 10 minutes).

  \ing[8]{oz}{cream cheese, at room temperature}
  \ing[1]{cup}{creamy peanut butter}
  \ing[1]{cup}{confectioner's sugar}
  \ing[\fr{1}{4}]{tsp}{kosher salt}

  In a large bowl, beat the cream cheese, peanut butter,
  confectioner's sugar, and salt with
  an electric mixer on medium speed until smooth
  and creamy (1 to 2 minutes), stopping to scrape down the sides of
  the bowl as needed.

  \ing[8]{oz}{thawed frozen whipped topping}

  Stir in the whipped topping until evenly distributed (about 1 minute).
  Spread the filling evenly over the cooled crust.

  \ing[13]{oz}{strawberry preserves}

  Spread the strawberry preserves evenly over the filling.

  \newstep

  Cover and chill until set (at least 4 hours or up to 12 hours).
  Slice into 9 equal portions and serve chilled.

\end{recipe}

\begin{recipe}{Lemony Raspberry Zucchini Bread with a Lemony Glaze}{}{}
  \newstep

  Preheat oven to 350 \0 F.
  Grease and flour a 9\X5-inch loaf pan; set aside.

  \ing[2]{cups}{unbleached all-purpose flour}
  \ing[2]{tsp}{baking powder}
  \ing[\fr{1}{2}]{tsp}{salt}

  In a large bowl, blend the flour, baking powder, and salt.

  \ing[2]{}{eggs}

  In the mixer bowl of a stand mixer (or a medium-sized bowl), beat 2 eggs well.

  \ing[\fr{1}{2}]{cup}{vegetable oil}
  \ing[\fr{2}{3}]{cup}{sugar}

  Add the oil and sugar, and beat on low until well combined.

  \ing[\fr{1}{2}]{cup}{whole milk}
  \ing[\fr{1}{4}]{cup}{fresh lemon juice}
  \Ing{zest of 1 lemon}

  Add the milk, lemon juice, and lemon zest, then blend everything well.

  \ing[1]{cup}{grated zucchini}

  Gently mix in the zucchini until just evenly distributed in the mixture.

  \newstep

  Add the dry ingredients and fold everything together, but do not overmix!

  \ing[1]{cup}{raspberries}

  Finally, fold in the raspberries.

  Pour the batter into the loaf pan.

  Bake for 45 minutes or until a toothpick inserted in the center
  comes out clean.
  Cool in the pan for 10 minutes, then remove to a wire rack and cool
  completely.

  \ing[1]{cup}{powdered sugar}
  \ing[1]{tbsp}{fresh lemon juice}

  While the loaf is cooling, in a small bowl, mix the powdered sugar
  and lemon juice until well blended.

  Spoon the glaze over the cooled loaf. Let the glaze set before
  slicing and serving.

\end{recipe}

\begin{recipe}{BEST Authentic Pfeffern\"usse}{}{}

  \ing[2\fr{1}{4}]{cup}{all-purpose flour}
  \ing[\fr{1}{2}]{tsp}{baking soda}
  \ing[\fr{1}{4}]{tsp}{salt}
  \ing[3--4]{tsp}{lebkuchengew\"urz}
  \ing[\fr{1}{4}]{tsp}{finely ground white pepper}
  \ing[\fr{1}{4}]{cup}{almond meal}

  In a small bowl, combine the flour, baking soda, salt,
  lebkuchengew\"urz, white pepper, and almond meal.

  Set aside.

  \ing[\fr{1}{2}]{cup}{packed brown sugar}
  \ing[\fr{1}{3}]{cup}{pure honey}
  \ing[5]{tbsp}{unsalted butter}
  \ing[3]{tbsp}{heavy cream}

  Combine the brown sugar, honey, butter, and cream in a medium
  saucepan. Heat, stirring frequently, until melted and the sugar is dissolved.
  Remove from heat and let sit for 5 minutes.

  \newstep
  Stir in the flour mixture.
  \ing[1]{}{large egg}
  Once incorporated, stir in the egg until thoroughly combined.
  The dough will have a nice, glossy sheen.
  It will be very sticky, which is how it should be.

  \newstep

  Turn the mixture out onto plastic wrap and wrap the dough tightly.
  Refrigerate overnight or for up to two days.

  \newstep

  Preheat the oven to 350\0 F.

  \newstep

  Remove the dough from the plastic wrap and immediately roll it into
  two \fr{3}{4}-inch thick strands.
  Slice the rolls into \fr{3}{4}-inch thick rounds and roll each
  round into a ball (each ball should be about \fr{3}{4} inch in diameter).
  Work quickly while the dough is still chilled.
  Note: Size is a matter of personal preference; you can make them
  larger if desired.

  \newstep

  Place the cookie balls on a lined cookie sheet and bake for about
  15 minutes, or until golden brown.
  Remove and let the cookies cool completely.

  \ing[2\fr{1}{2}]{cup}{powdered sugar}
  \ing[3--4]{tbsp}{hot water}

  To make the glaze, combine the powdered sugar and hot water until smooth.

  \newstep

  Dip each cookie in the glaze, letting the excess drip off. Place
  them on a wire rack positioned over a cookie sheet (to catch drips)
  and let them sit until the glaze is fully hardened.

  \newstep

  Store the cookies in an airtight container in a cool place.
  They will keep for at least 2 weeks (they are usually stored
  longer), and the flavor only improves over time.
\end{recipe}

\begin{recipe}{Authentic Lebkuchengev\"urz (German Gingerbread Spice)}{}{}

  \ing[2\fr{1}{2}]{tbsp}{ground cinnamon}
  \ing[2]{tsp}{ground cloves}
  \ing[\fr{1}{2}]{tsp}{ground allspice}
  \ing[\fr{1}{2}]{tsp}{ground coriander}
  \ing[\fr{1}{2}]{tsp}{ground green cardamom}
  \ing[\fr{1}{2}]{tsp}{ground ginger}
  \ing[\fr{1}{2}]{tsp}{ground star anise}
  \ing[\fr{1}{4}]{tsp}{ground mace}
  \ing[\fr{1}{4}]{tsp}{ground nutmeg}

  Combine the spices and store in an airtight jar in a cool, dark
  place for up to one year.

  \freeform

  For superior flavor, grind these spices from toasted and freshly
  ground whole spices.
  Simply heat a dry skillet over medium heat and toast the whole
  spices until very fragrant.
  Be careful not to scorch the spices, or they will become bitter.
  Using the freshly ground blend within a few days is optimal.

\end{recipe}
